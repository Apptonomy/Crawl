# Crawl
A travel app for people who love to drink

# Install dependencies
npm install

# Install Client dependencies
cd client
npm install

# Serve client
npm run client(in client)

# Drop then add the database
npm run db (in root)

# Run server
VScode debugger or npm run server (in client)

# Production Build
cd client
npm run build

# Enjoy
